const getBlobFromUrl = (myImageUrl) => {
  return new Promise((resolve, reject) => {
      let request = new XMLHttpRequest();
      request.open('GET', myImageUrl, true);
      request.responseType = 'blob';
      request.onload = () => {
          resolve(request.response);
      };
      request.onerror = reject;
      request.send();
  })
}
