clipboard.js

variant.js 中針對標題URL的複製。(以下內容已經不再目前版本，也可以在之前的learn.js中找到)

jQuery(document).ready(function() {
    // 利用ClipboardJS 將類別anchor 的作為複製對象,這個對象要有
    // data-clipboard-text,也就是要複製甚麼。
    var text, clip = new ClipboardJS('.anchor');
    //append 插入h1的子代,不是同輩
    $("h1~h2,h1~h3,h1~h4,h1~h5,h1~h6").append(function(index, html){
      var element = $(this);
      //mnnote: encodeURI why?
      //var url = encodeURI(document.location.origin + document.location.pathname);
      // 下面也可以=window.location.href
      //var url = document.location.origin + document.location.pathname;
      
      var link = window.location.href + "#"+element[0].id;
      return " <span class='anchor' data-clipboard-text='"+link+"'>" +
        "<i class='fas fa-link fa-lg'></i>" +
        "</span>"
      ;
    });
  });

版面相關

  • 文字背景顏色對調
  • 解釋css display，最佳測試✔地點
    • block
    • inline :　直接當成一個文字使用，但是如果太大，會覆蓋其他上下行文字（也就是重疊），如果不想重疊，可以設定line-height（例如line-height:90px`(>圖片大小)此時,就會撐開這一行,而不會和上下行重疊。
    • inline-block: 上面有些css屬性無法設定,這裡就是綜合的意思。
    • table-cell: 通常用來配合父元素設定為table,(通常用來作為版面設定)
    • list-item
    • none

參考

  • X Y : X節點下的所有子代Y
  • X > Y: X節點下的第一層子代的Y
  • X + Y: X節點之後同層且第一個Y
  • X ~ Y: X節點之後同層的Y

以後出現的症狀

症狀

⁉ 出現雙條垂直捲動軸

💡 解法: theme.css

body {
    overflow:hidden;
}

⁉ #top-bar 會隨著移動不見

💡 解法

# top-bar {
position: fixed;
top 3px;
}

⁉ #top-bar 會遮住最頂層的文字

💡 解法:暴力解,直接就空掉最前面的版面,讓給#top-bar 。

#body{
  margin-top: 3rem;
}

⁉ 無法隱藏左側藍 💡 解法

  • #sidebar-toggle-span 按這個顯示/隱藏切換

  • #sidebar 就左側面板。

    $("#sidebar-toggle-span").click(
  function(){
    if( $("#sidebar").css("display")=="block")
      $("#sidebar").css("display","none");    
    else
    $("#sidebar").css("display","block");  
  }
);
